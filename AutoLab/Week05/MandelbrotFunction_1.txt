from multiprocessing.pool import ThreadPool

def generate_mandelbrot_set(points, num_processes):
    def escape(c):
        z = 0
        for i in range(100):
            z = z**2 + c
            if np.abs(z) > 2.0:
                return i
        return 100

    with multiprocessing.pool.ThreadPool(num_processes) as pool:
        escape_times = np.array(pool.map(escape, points))

    return escape_times
